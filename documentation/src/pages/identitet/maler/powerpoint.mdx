export const query = graphql`
query fargerQuery {
  files: allFile(filter: {sourceInstanceName: {eq: "media"}, relativeDirectory:  {glob: "images/identitet/maler"}, extension: {eq: "png"}}) {
    images: nodes {
      name
       childImageSharp {
         gatsbyImageData(
          layout: CONSTRAINED
        )
      }
    }
  }
}
`;
