import Tooltip from '@entur/tooltip';
import '@entur/tooltip/dist/styles.css';
import { Props } from 'docz';
import { Playground } from '../../components/Playground';
import PageHeader from '../../components/PageHeader';

<PageHeader>
  Tooltip kan brukes for å gi ekstra informasjon til brukeren ved hover eller
  fokus.
</PageHeader>

Man må derimot _unngå_ å legge nødvendig og kritisk informasjon i tooltip'en.

## Props

<Props of={Tooltip} />

## Eksempel

<Playground>
  <Tooltip content="Her er tooltipen" placement="top">
    Denne teksten har en tooltip
  </Tooltip>
</Playground>

Ettersom tooltip _skal_ brukes på fokuserbare elementer, så vises også tooltip om knappen/linken etc er fokusert

<Playground>
  <Tooltip content="Tooltip til knappen" placement="top">
    <button>Her er en knapp med tooltip</button>
  </Tooltip>
</Playground>

Tooltip-teksten vil også bli lagt til i trigger-elementet som en aria-label.
